Browser Mode:

# Start UI server
npm run ui

# Open in browser
http://localhost:5173/ui/watch.html

# Or specify a file
http://localhost:5173/ui/watch.html?file=src/strudel/hello.str

OSC Mode (TidalCycles style):

# 1. Start SuperCollider and run:
SuperDirt.start

# 2. Run OSC watcher (in another terminal)
npm run osc src/strudel/hello.str

# 3. Edit src/strudel/hello.str in your editor
# Changes are automatically sent to SuperDirt!
